## 1) Set up Python environment (run from repo root)
python -m venv .venv
# PowerShell:
 .\.venv\Scripts\Activate.ps1
# Git Bash:
 source .venv/Scripts/activate
pip install -r api/requirements.txt

## 2) Start backend (choose ONE)
# Option A: Uvicorn with auto-reload (recommended during development)
python -m uvicorn api.main:app --host 0.0.0.0 --port 8000 --reload

# Option B: Dev runner (no auto-reload, robust from any CWD)
# From repo root:
python api/run_dev.py
# Or if you're inside the api folder:
python run_dev.py

## 3) Start frontend
cd frontend
python -m http.server 5500

## 4) Verify locally
# Backend health
# Open in browser:
http://localhost:8000/api/health
# FastAPI docs:
http://localhost:8000/docs

# Frontend app
http://localhost:5500/index.html

## Notes
# - Run commands from repo root unless noted.
# - In Git Bash, use forward slashes in paths (e.g., api/run_dev.py).
# - If you change backend port, pass --port or set PORT in .env.
